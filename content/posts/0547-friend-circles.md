/**
 * @param {number[][]} M
 * @return {number}
 */
var findCircleNum = function(M) {
  let count = 0
  const len = M.length,
    visited = new Array(len).fill(0)
  const dfs = row => {
    for (let col = 0; col < len; col++) {
      const person = M[row][col]
      if (person === 1 && visited[col] === 0) {
        visited[col] = 1
        dfs(col)
      }
    }
  }

  for (let row = 0; row < len; row++) {
    if (visited[row] === 0) {
      dfs(row)
      count++
    }
  }
  return count
}
