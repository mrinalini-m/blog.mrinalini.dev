 * @param {number} K
 * @return {number}
 */

//iterative O(n^2) runtime. Won't pass leetcode 'Status: Time Limit Exceeded'
var subarraysDivByK = function(A, K) {
  let count = 0
  const sum = []
  A.forEach(curr => {
    sum.push(0)
    sum.forEach((elem, j) => {
      sum[j] = curr + elem
      if (sum[j] % K === 0 || sum[j] % K === -0) {
        count++
      }
    })
  })

  return count
}
