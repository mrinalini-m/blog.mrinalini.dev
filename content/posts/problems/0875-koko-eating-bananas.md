// /**
//  * @param {number[]} piles
//  * @param {number} H
//  * @return {number}
//  */

var minEatingSpeed = function(piles, H) {
  const isPossibleK = k => {
    let t = 0
    for (const pile of piles) {
      t += Math.ceil(pile / k)
    }
    return t <= H
  }

  let max = Math.max(...piles)
  min = 0

  while (min < max) {
    mid = Math.floor((min + max) / 2)
    if (!isPossibleK(mid)) {
      min = mid + 1
    } else max = mid
  }
  return min
}

  H = 5
