class Trie {
	constructor() {
		this.letters = {}
	}

	insert(word) {
		const len = word.length
		let curr = this.letters
		for (let i = 0; i < len; i++) {
			const letter = word[i]
			if (!curr[letter]) curr[letter] = {}
			curr = curr[letter]
		}
		curr.isEnd = true
	}

	traverse(word) {
		const len = word.length
		let curr = this.letters
		for (let i = 0; i < len; i++) {
			const letter = word[i]
			if (!curr[letter]) return false
			curr = curr[letter]
		}
		return curr
	}

	search(word) {
		return !!this.traverse(word).isEnd
	}

	startsWith(prefix) {
		return !!this.traverse(prefix)
	}
}
