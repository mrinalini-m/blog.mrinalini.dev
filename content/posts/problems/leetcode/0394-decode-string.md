function decodeString(str) {
	const stack = []
	let subStr = '',
		multiplier = ''
	for (curr of str) {
		if (curr === '[') {
			stack.push(subStr)
			stack.push(parseInt(multiplier))
			multiplier = ''
			subStr = ''
		} else if (curr === ']') {
			multiplier = stack.pop()
			const prevStr = stack.pop()
			subStr = prevStr + subStr.repeat(multiplier)
			multiplier = ''
		} else if (Number.isInteger(parseInt(curr))) {
			multiplier += curr
		} else {
			subStr += curr
		}
	}

	return subStr
}
